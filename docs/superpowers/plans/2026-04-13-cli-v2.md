```typescript
// src/routes/api/pub/[id]/resolve/+server.ts
import { json, error } from '@sveltejs/kit';
import type { RequestHandler } from './$types';
import { getDb } from '$lib/server/db';

export const POST: RequestHandler = async ({ params, request, platform }) => {
  if (!platform) throw error(500, 'No platform');
  const db = getDb(platform);
  const body = await request.json();

  if (body.all) {
    await db
      .prepare('UPDATE comments SET resolved = 1 WHERE page_id = ? AND resolved = 0')
      .bind(params.id)
      .run();
  } else if (body.comment_ids?.length) {
    const placeholders = body.comment_ids.map(() => '?').join(',');
    await db
      .prepare(`UPDATE comments SET resolved = 1 WHERE page_id = ? AND id IN (${placeholders})`)
      .bind(params.id, ...body.comment_ids)
      .run();
  } else {
    throw error(400, 'Provide comment_ids array or all: true');
  }

  return json({ ok: true });
};
```

```javascript
import { getBaseUrl, getToken } from './config.js';

function headers() {
  const h = { 'Content-Type': 'application/json' };
  const token = getToken();
  if (token) h['Cookie'] = `vibe_session=${token}`;
  return h;
}

async function request(method, path, body) {
  const res = await fetch(`${getBaseUrl()}${path}`, {
    method,
    headers: headers(),
    body: body ? JSON.stringify(body) : undefined,
  });
  if (!res.ok) {
    const text = await res.text();
    let msg;
    try { msg = JSON.parse(text).message; } catch { msg = text; }
    throw Object.assign(new Error(msg || `HTTP ${res.status}`), { status: res.status });
  }
  if (res.status === 204) return null;
  return res.json();
}

export async function publish(markdown, options = {}) {

}
```

Replace `cli/bin/vibe.js` entirely:

```javascript
#!/usr/bin/env node
import { readFileSync } from 'fs';
import * as api from '../lib/api.js';
import { saveConfig, getToken, getBaseUrl } from '../lib/config.js';
import { out, err } from '../lib/output.js';

const args = process.argv.slice(2);

// Check for --mcp flag first
if (args.includes('--mcp')) {
  const { startMcp } = await import('./mcp.js');
  await startMcp();
  process.exit(0);
}

const cmd = args[0];
const format = args.includes('--format') ? args[args.indexOf('--format') + 1] : 'json';

function parseFlags(argv) {
  const flags = {};
  for (let i = 0; i < argv.length; i++) {
    if (argv[i].startsWith('--') && argv[i] !== '--format') {
      flags[argv[i].slice(2)] = argv[i + 1] ?? true;
      i++;
    }
  }
  return flags;
}

async function readStdin() {
  return new Promise((resolve) => {
    let data = '';
    process.stdin.setEncoding('utf8');
    process.stdin.on('data', (chunk) => (data += chunk));
    process.stdin.on('end', () => resolve(data));
  });
}

function readFile(path) {
  try { return readFileSync(path, 'utf8'); } catch { err(`Could not read file: ${path}`); }
}

function getMarkdown(fileArg) {
  if (fileArg && !fileArg.startsWith('--')) return readFile(fileArg);
  return null;
}

async function resolveSlug(slug) {
  const page = await api.getBySlug(slug);
  return page;
}

async function main() {
  try {
    if (!cmd || cmd === 'help' || cmd === '-h' || cmd === '--help') {
      out({ commands: ['publish','get','list','update','delete','comments','comment','resolve','versions','version','login','config','whoami'] }, format);
      return;
    }

    if (cmd === 'publish' || cmd === 'pub') {
      const fileArg = args[1] && !args[1].startsWith('--') ? args[1] : null;
      const flags = parseFlags(fileArg ? args.slice(2) : args.slice(1));
      const markdown = getMarkdown(fileArg) ?? await readStdin();
      if (!markdown?.trim()) err('No markdown content');
      const result = await api.publish(markdown, flags);
      out(result, format);
      return;
    }

    if (cmd === 'get') {
      const slug = args[1];
      if (!slug) err('Usage: vibe-pub get <slug>');
      const page = await api.getBySlug(slug);
      out(page, format);
      return;
    }

    if (cmd === 'list' || cmd === 'ls') {
      if (!getToken()) err('Not logged in. Run: vibe-pub config --token <token>');
      const pages = await api.list();
      out(pages, format);
      return;
    }

    if (cmd === 'update') {
      const slug = args[1];
      if (!slug) err('Usage: vibe-pub update <slug> [file.md]');
      const fileArg = args[2] && !args[2].startsWith('--') ? args[2] : null;
      const markdown = getMarkdown(fileArg) ?? await readStdin();
      if (!markdown?.trim()) err('No markdown content');
      const page = await resolveSlug(slug);
      const result = await api.update(page.id, markdown);
      out(result, format);
      return;
    }

    if (cmd === 'delete' || cmd === 'rm') {
      const slug = args[1];
      if (!slug) err('Usage: vibe-pub delete <slug>');
      const page = await resolveSlug(slug);
      await api.remove(page.id);
      out({ deleted: slug }, format);
      return;
    }

    if (cmd === 'comments') {
      const slug = args[1];
      if (!slug) err('Usage: vibe-pub comments <slug>');
      const page = await resolveSlug(slug);
      const comments = await api.getComments(page.id);
      out(comments, format);
      return;
    }

    if (cmd === 'comment') {
      const slug = args[1];
      const body = args[2];
      if (!slug || !body) err('Usage: vibe-pub comment <slug> "body"');
      const flags = parseFlags(args.slice(3));
      const page = await resolveSlug(slug);
      const comment = await api.addComment(page.id, body, {
        anchor: flags.anchor ? { type: 'block', block_id: flags.anchor } : undefined,
      });
      out(comment, format);
      return;
    }

    if (cmd === 'resolve') {
      const slug = args[1];
      if (!slug) err('Usage: vibe-pub resolve <slug> [--all | --ids id1,id2]');
      const flags = parseFlags(args.slice(2));
      const page = await resolveSlug(slug);
      const ids = flags.ids ? flags.ids.split(',') : undefined;
      const result = await api.resolveComments(page.id, { all: !!flags.all, ids });
      out(result, format);
      return;
    }

    if (cmd === 'versions') {
      const slug = args[1];
      if (!slug) err('Usage: vibe-pub versions <slug>');
      const page = await resolveSlug(slug);
      const versions = await api.getVersions(page.id);
      out(versions, format);
      return;
    }

    if (cmd === 'version') {
      const slug = args[1];
      const num = args[2];
      if (!slug || !num) err('Usage: vibe-pub version <slug> <number>');
      const page = await resolveSlug(slug);
      const version = await api.getVersion(page.id, num);
      out(version, format);
      return;
    }

    if (cmd === 'whoami') {
      if (!getToken()) err('Not logged in');
      out({ token: '***', baseUrl: getBaseUrl() }, format);
      return;
    }

    if (cmd === 'login') {
      const email = args[1];
      if (!email) err('Usage: vibe-pub login <email>');
      out({ action: 'login', steps: [`Visit ${getBaseUrl()}/auth/login`, `Sign in with ${email}`, 'Copy vibe_session cookie', 'Run: vibe-pub config --token <token>'] }, format);
      return;
    }

    if (cmd === 'config') {
      const flags = parseFlags(args.slice(1));
      if (flags.token) { saveConfig({ token: flags.token }); out({ saved: 'token' }, format); }
      else if (flags['base-url']) { saveConfig({ baseUrl: flags['base-url'] }); out({ saved: 'baseUrl', value: flags['base-url'] }, format); }
      else err('Usage: vibe-pub config --token <t> or --base-url <u>');
      return;
    }

    err(`Unknown command: ${cmd}`);
  } catch (e) {
    err(e.message, e.status ?? 1);
  }
}

main();
```

```javascript
import * as api from '../lib/api.js';

const TOOLS = [
  { name: 'publish', description: 'Publish markdown to vibe.pub', inputSchema: { type: 'object', properties: { markdown: { type: 'string', description: 'Markdown content' }, slug: { type: 'string' }, view: { type: 'string', enum: ['doc','kanban','changelog','timeline','slides','dashboard'] }, access: { type: 'string', enum: ['public','unlisted','private'] }, theme: { type: 'string' } }, required: ['markdown'] } },
  { name: 'get_page', description: 'Get page by slug', inputSchema: { type: 'object', properties: { slug: { type: 'string' } }, required: ['slug'] } },
  { name: 'update_page', description: 'Update page content (auto-snapshots old version)', inputSchema: { type: 'object', properties: { slug: { type: 'string' }, markdown: { type: 'string' } }, required: ['slug', 'markdown'] } },
  { name: 'delete_page', description: 'Delete a page', inputSchema: { type: 'object', properties: { slug: { type: 'string' } }, required: ['slug'] } },
  { name: 'list_pages', description: 'List your pages (requires auth)', inputSchema: { type: 'object', properties: {} } },
  { name: 'get_comments', description: 'Get comments on a page', inputSchema: { type: 'object', properties: { slug: { type: 'string' } }, required: ['slug'] } },
  { name: 'add_comment', description: 'Add comment to a page', inputSchema: { type: 'object', properties: { slug: { type: 'string' }, body: { type: 'string' }, anchor: { type: 'string', description: 'Block ID to anchor comment to' } }, required: ['slug', 'body'] } },
  { name: 'resolve_comments', description: 'Mark comments as resolved', inputSchema: { type: 'object', properties: { slug: { type: 'string' }, all: { type: 'boolean' }, comment_ids: { type: 'array', items: { type: 'string' } } }, required: ['slug'] } },
  { name: 'get_versions', description: 'Get version history', inputSchema: { type: 'object', properties: { slug: { type: 'string' } }, required: ['slug'] } },
  { name: 'get_version', description: 'Get specific version content', inputSchema: { type: 'object', properties: { slug: { type: 'string' }, version: { type: 'number' } }, required: ['slug', 'version'] } },
];

async function resolveSlug(slug) {
  return api.getBySlug(slug);
}

async function handleTool(name, args) {
  switch (name) {
    case 'publish': return api.publish(args.markdown, args);
    case 'get_page': return api.getBySlug(args.slug);
    case 'update_page': { const p = await resolveSlug(args.slug); return api.update(p.id, args.markdown); }
    case 'delete_page': { const p = await resolveSlug(args.slug); await api.remove(p.id); return { deleted: args.slug }; }
    case 'list_pages': return api.list();
    case 'get_comments': { const p = await resolveSlug(args.slug); return api.getComments(p.id); }
    case 'add_comment': { const p = await resolveSlug(args.slug); return api.addComment(p.id, args.body, { anchor: args.anchor ? { type: 'block', block_id: args.anchor } : undefined }); }
    case 'resolve_comments': { const p = await resolveSlug(args.slug); return api.resolveComments(p.id, { all: args.all, ids: args.comment_ids }); }
    case 'get_versions': { const p = await resolveSlug(args.slug); return api.getVersions(p.id); }
    case 'get_version': { const p = await resolveSlug(args.slug); return api.getVersion(p.id, args.version); }
    default: throw new Error(`Unknown tool: ${name}`);
  }
}

export async function startMcp() {
  const server = new Server({ name: 'vibe-pub', version: '0.2.0' }, { capabilities: { tools: {} } });

  server.setRequestHandler(ListToolsRequestSchema, async () => ({ tools: TOOLS }));

  server.setRequestHandler(CallToolRequestSchema, async (request) => {
    try {
      const result = await handleTool(request.params.name, request.params.arguments ?? {});
      return { content: [{ type: 'text', text: JSON.stringify(result, null, 2) }] };
    } catch (e) {
      return { content: [{ type: 'text', text: JSON.stringify({ error: e.message }) }], isError: true };
    }
  });

  const transport = new StdioServerTransport();
  await server.connect(transport);
}
```

```typescript
import { test, expect } from '@playwright/test';

test.describe('Agent workflow: publish → comment → regen → resolve', () => {
  test('full cycle', async ({ request }) => {
    // 1. Publish
    const pub = await (await request.post('/api/pub', {
      data: { markdown: '# Agent Report\n\n## Security\n\nJWT tokens expire after 7 days.' },
    })).json();
    expect(pub.slug).toBeTruthy();

    // 2. Human adds comment
    const comment = await (await request.post(`/api/comment/${pub.id}`, {
      data: { body: 'Should be 24 hours, not 7 days', display_name: 'reviewer' },
    })).json();
    expect(comment.id).toBeTruthy();

    // 3. Agent reads comments
    const comments = await (await request.get(`/api/comment/${pub.id}`)).json();
    expect(comments.length).toBe(1);
    expect(comments[0].body).toContain('24 hours');

    // 4. Agent regenerates with feedback incorporated
    const updateRes = await request.put(`/api/pub/${pub.id}`, {
      data: { markdown: '# Agent Report\n\n## Security\n\nJWT tokens expire after 24 hours.' },
    });
    expect(updateRes.status()).toBe(200);

    // 5. Verify version was created
    const versions = await (await request.get(`/api/pub/${pub.id}/versions`)).json();
    expect(versions.length).toBe(1);
    expect(versions[0].version).toBe(1);

    // 6. Verify old version has original content
    const v1 = await (await request.get(`/api/pub/${pub.id}/versions/1`)).json();
    expect(v1.markdown).toContain('7 days');

    // 7. Agent resolves the comment
    const resolveRes = await request.post(`/api/pub/${pub.id}/resolve`, {
      data: { all: true },
    });
    expect(resolveRes.status()).toBe(200);

    // 8. Verify comment is resolved
    const finalComments = await (await request.get(`/api/comment/${pub.id}`)).json();
    expect(finalComments[0].resolved).toBe(1);

    // 9. Verify current page has updated content
    const page = await (await request.get(`/api/pub/by-slug/${pub.slug}`)).json();
    expect(page.markdown).toContain('24 hours');
  });
});
```
